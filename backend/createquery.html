CreateQuery


String nombreDepartamento="Software";
String queryString = "SELECT COUNT(e) FROM Estudiante e join e.tutorias t WHERE t.profesor.departamento.nombre LIKE :parametro1";
Query query = em.createQuery(queryString);
query.setParameter("parametro1", "%"+ nombreDepartamento);
		
System.out.println("Numero de estudiantes tutoriados por Dep. de Software: "+ query.getSingleResult());
